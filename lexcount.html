%option noyywrap
%{
 int lcount = 0,Ccount = 0,Wcount=0;
%}

%%
. { Ccount++; } 
\n { lcount++; Ccount++; } 
([a-zA-Z0-9])* { Wcount++; }
%%

void main()
{
	printf("Enter your input : ");
	yylex();
	printf("\nValue of lines count : %d\n",lcount);
	printf("\nValue of character count : %d\n",Ccount);
	printf("\nValue of word count : %d\n",Wcount);
}
